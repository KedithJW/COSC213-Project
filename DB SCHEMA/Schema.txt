Table User {
  user_id integer [pk, increment]
  email varchar(255) [not null, unique]
  hashed_password varchar(255) [not null]
  f_name varchar(100)
  l_name varchar(100)
  phone varchar(25)
  created_at datetime [default: 'now()']
}

Table Project {
  project_id integer [pk, increment]
  name varchar(200) [not null]
  owner_user_id int [not null, ref: > User.user_id]
  created_at datetime [default: 'now()']

  Indexes {
    name
  }
}

Table Board {
  board_id int [pk, increment]
  project_id int [not null, ref: > Project.project_id]
  name varchar(200) [not null]
  created_at datetime [default: 'now()']

  Indexes {
    (project_id, name) [unique]
     project_id
  }
}


Table Task {
 task_id integer [pk, increment]
 board_id int [not null, ref: > Board.board_id]
 name varchar(255) [not null]
 completed boolean [not null, default: false]
 created_at datetime [default: 'now()']

 Indexes {
   board_id
  (board_id, completed)
 }
}
